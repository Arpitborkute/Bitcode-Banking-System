import java.io.IOException;
import java.util.Comparator;


public class Bitcode{
        public static void main(String[] args) throws IOException {
            Bank bank = new Bank();
            bank.startBackup();

            SavingsAccount savingsAccount = new SavingsAccount("1110", "Krish", 3050, "krin@upi", 2.5);
            bank.addAccount(savingsAccount);

            CurrentAccount currentAccount = new CurrentAccount("34454", "Raj", 5000, "raj@upi");
            bank.addAccount(currentAccount);

            savingsAccount.deposit(500);
            savingsAccount.applyInterest();
            savingsAccount.viewStatement();
            savingsAccount.saveStatementToFile();

            currentAccount.withdraw(700);
            currentAccount.viewStatement();

            bank.viewAccountsSorted(Comparator.comparing(acc -> acc.accountHolderName));
        }
    }
